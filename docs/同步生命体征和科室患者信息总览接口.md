<soapenv:Envelope xmlns:soapenv="http://schemas.xmlsoap.org/soap/envelope/" xmlns:i="http://i.sync.common.pkuih.iih/">
   <soapenv:Header/>
   <soapenv:Body>
      <i:getEntVt>
         <!--Optional:-->
         <i:arg0>2026-04-01 13:04:46</i:arg0>
         <!--Optional:-->
